\section*{Boundary Conditions - Theory 1}

\subsection*{Description}
A vertical force $F$ is applied at the handle end ($x = x_F = 900$ mm). The oar is clamped at the oarlock position for all $x \leq 0$.

\subsection*{Mathematical Formulation}

\begin{tabular}{lll}
\hline
\textbf{Location} & \textbf{Condition} & \textbf{Description} \\
\hline
$x = x_F$ & $V(x_F) = -F$ & Applied force (shear force) \\
$x = x_F$ & $M(x_F) = 0$ & Free end (no moment) \\
\hline
$x = 0$ & $w(0) = 0$ & No vertical displacement \\
$x = 0$ & $\theta(0) = 0$ & No rotation (clamped) \\
\hline
$x \leq 0$ & Fixed & Clamped region \\
\hline
\end{tabular}

\vspace{5mm}
\noindent where:
\begin{itemize}
\item $F$ = applied vertical force at handle [N]
\item $V(x)$ = shear force at position $x$ [N]
\item $M(x)$ = bending moment at position $x$ [NÂ·mm]
\item $w(x)$ = vertical deflection at position $x$ [mm]
\item $\theta(x)$ = rotation angle at position $x$ [rad]
\item $x_F = 900$ mm = handle position
\end{itemize}
